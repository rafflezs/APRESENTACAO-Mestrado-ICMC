\begin{frame}{Factibilidade do NSGA-II}

    \setbeamercolor{block title}{fg=white, bg=red!40}

    \begin{block}{Incongruência comparativa}
        NSGA-II não satisfaz (R\ref{eq:r6}) \(\rightarrow\) \textbf{não comparável diretamente} ao MIP.
    \end{block}
    \endgroup

    \vspace{0.5cm}

    \begin{itemize}
        \item Estruturais R\ref{eq:r1}--R\ref{eq:r4}: satisfeitas (Estendido 100\%, Compacto 89,8\%).
        \item Cobertura mínima violada em 4--6\% dos períodos (turno vespertino).
        \item Compacto: 10,2\% violam capacidade de sala (R\ref{eq:r5}, clínica U06).
    \end{itemize}

    {\footnotesize Encaminhamento: tendências e \textit{trade-offs} $\rightarrow$ \textbf{operador de reparo} (trabalho futuro).}

\end{frame}

%%% Frame 5.7 — Discussão: síntese
\begin{frame}{Sumarizando}
    \begin{itemize}
        \item[\textbf{1.}] \textbf{Exato} escala mal: ótimo em 15 d; $\geq$30 d satura o tempo (gap $>40\%$).
        \item[\textbf{2.}] \textbf{Híbrido} supera RF em horizontes curtos; RF puro é 30--55\% mais rápido.
        \item[\textbf{3.}] \textbf{NSGA-IIc} domina o Estendido; ``supera'' o exato em 60 d apenas por referência sub-ótima.
        \item[\textbf{4.}] \textbf{Factibilidade} parcial do NSGA-II aponta o próximo passo: reparo de cobertura.
    \end{itemize}
\end{frame}
